#define lowbit(x) ((x) & -(x))
inline void setmax(int& x, int y) { if(y > x) x = y; }

int n, A[N], bit[N];

// max(A[1], ..., A[i])
inline int prefixMax(int i)
{
    int res = -INF;
    for(; i>0; i-=lowbit(i))
        setmax(res, bit[i]);
    return res;
}

// A[i] = max(A[i], val)
inline void update(int i, int val)
{
    for(; i<=n; i+=lowbit(i))
        setmax(bit[i], val);
}
