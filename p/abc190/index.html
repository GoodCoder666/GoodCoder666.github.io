#include <cstdio>
#include <cstring>
using namespace std;

int a[105], b[105], c[20], d[20];

int main()
{
	int n, m, k;
	scanf("%d%d", &n, &m);
	for(int i=0; i<m; i++)
	{
		scanf("%d%d", a + i, b + i);
		a[i] --, b[i] --;
	}
	scanf("%d", &k);
	for(int i=0; i<k; i++)
	{
		scanf("%d%d", c + i, d + i);
		c[i] --, d[i] --;
	}
	int limit = 1 << k, ans = 0;
	for(int st=0; st<limit; st++)
	{
		bool hasdish[105] = {false};
		for(int i=0; i<k; i++)
			if(st & (1 << i))
				hasdish[c[i]] = true;
			else hasdish[d[i]] = true;
		int cnt = 0;
		for(int i=0; i<m; i++)
			cnt += hasdish[a[i]] && hasdish[b[i]];
		if(cnt > ans) ans = cnt;
	}
	printf("%d\n", ans);
	return 0;
}
