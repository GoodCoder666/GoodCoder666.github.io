#include <cstdio>
#include <queue>
using namespace std;

typedef long long LL;

int main(int argc, char** argv)
{
    int k;
    scanf("%d", &k);
    queue<LL> q;
    for(int i=1; i<10; i++) q.push(i);
    while(true)
    {
        LL x = q.front(); q.pop();
        if(--k == 0)
        {
            printf("%lld\n", x);
            return 0;
        }
        int r = x % 10LL;
        x *= 10LL;
        x += r;
        if(r > 0) q.push(x - 1);
        q.push(x);
        if(r < 9) q.push(x + 1);
    }
    return 0;
}
