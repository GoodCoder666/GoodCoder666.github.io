#include <cstdio>
#define maxn 200005
using namespace std;

bool a[maxn];

int main()
{
    int n = 0;
    char c;
    while((c = getchar()) != '\n')
        n = (n << 3) + (n << 1) + (c ^ 48);
    for(int i=0; i<n; i++, getchar())
        a[i] = getchar() ^ 48;
    bool flipped = false;
    for(int l=0, r=n-1; a[l]==flipped; flipped^=1, l++)
        while(!a[r] ^ flipped)
            if(l > --r)
                return puts("Yes"), 0;
    puts("No");
    return 0;
}
