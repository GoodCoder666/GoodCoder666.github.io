#include <cstdio>
#include <set>
#include <vector>
using namespace std;

int cnt[10];

int f(int x)
{
    for(int i=0; i<10; i++) cnt[i] = 0;
    while(x > 0)
    {
        cnt[x % 10] ++;
        x /= 10;
    }
    int g1 = 0, g2 = 0, t = 1;
    for(int i=0; i<10; i++)
        while(cnt[i]--)
        {
            g1 += i * t, g2 = g2 * 10 + i;
            t *= 10;
        }
    return g1 - g2;
}

int main()
{
    int n, k;
    scanf("%d%d", &n, &k);
    while(k--) n = f(n);
    printf("%d\n", n);
    return 0;
}

#include <cstdio>
using namespace std;

typedef unsigned long long ULL;

char x[65];
ULL m;

inline void setmax(int& x, int y)
{
    if(y > x) x = y;
}

inline bool check(const ULL& base)
{
    // Returns: (x -> base) <= m?
    ULL t = 0ULL;
    for(int i=0; x[i]; i++)
    {
        if(t > m / base)
            return false;
        t *= base;
        if((t += x[i] - '0') > m)
            return false;
    }
    return true;
}

int main()
{
    scanf("%s%llu", x, &m);
    int d = 0;
    for(int i=0; x[i]; i++)
        setmax(d, x[i] - '0');
    if(x[1] == '\0')
    {
        puts(d > m? "0": "1");
        return 0;
    }
    ULL l = d, r = m;
    while(l < r)
    {
        ULL mid = l + r + 1ULL >> 1ULL;
        if(check(mid)) l = mid;
        else r = mid - 1ULL;
    }
    printf("%llu\n", l - d);
    return 0;
}
