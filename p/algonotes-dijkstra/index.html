#include <cstdio>
#include <queue>
#define maxn 100005
#define INF 9223372036854775807LL
using namespace std;

using LL = long long;
using pli = pair<LL, int>;

vector<pair<int, int>> G[maxn];
LL dis[maxn];
int par[maxn];

int main()
{
    int n, m;
    scanf("%d%d", &n, &m);
    while(m--)
    {
        int u, v, c;
        scanf("%d%d%d", &u, &v, &c);
        G[--u].emplace_back(--v, c);
        G[v].emplace_back(u, c);
    }
    for(int i=1; i<n; i++) dis[i] = INF;
    par[0] = -1;
    priority_queue<pli, vector<pli>, greater<pli>> q;
    q.emplace(0LL, 0);
    while(!q.empty())
    {
        auto [d, v] = q.top(); q.pop();
        if(dis[v] == d)
            for(auto [u, w]: G[v])
                if(d + w < dis[u])
                    par[u] = v,
                    q.emplace(dis[u] = d + w, u);
    }
    if(dis[n - 1] == INF) { puts("-1"); return 0; }
    vector<int> path;
    int v = n - 1;
    while(v != -1)
    {
        path.push_back(v);
        v = par[v];
    }
    for(int i=path.size()-1; i>=0; i--)
        printf("%d ", ++path[i]);
    return 0;
}

#include <set>
#define maxn 100005
#define INF 9223372036854775807LL
using namespace std;

using LL = long long;
using pli = pair<LL, int>;

vector<pair<int, int>> G[maxn];
LL dis[maxn];
int par[maxn];

int main()
{
    int n, m;
    scanf("%d%d", &n, &m);
    while(m--)
    {
        int u, v, c;
        scanf("%d%d%d", &u, &v, &c);
        G[--u].emplace_back(--v, c);
        G[v].emplace_back(u, c);
    }
    for(int i=1; i<n; i++) dis[i] = INF;
    par[0] = -1;
    set<pli> s;
    s.emplace(0LL, 0);
    while(!s.empty())
    {
        auto it = s.begin(); s.erase(it);
        auto [d, v] = *it;
        for(auto [u, w]: G[v])
            if(d + w < dis[u])
            {
                par[u] = v;
                if(dis[u] != INF)
                    s.erase(pli(dis[u], u));
                s.emplace(dis[u] = d + w, u);
            }
    }
    if(dis[n - 1] == INF) { puts("-1"); return 0; }
    vector<int> path;
    int v = n - 1;
    while(v != -1)
    {
        path.push_back(v);
        v = par[v];
    }
    for(int i=path.size()-1; i>=0; i--)
        printf("%d ", ++path[i]);
    return 0;
}
