#include <cstdio>
#define maxn 200005
using namespace std;

inline void setmin(int& x, int y)
{
    if(y < x) x = y;
}

int a[maxn], pre[maxn], suf[maxn];

int main()
{
    int n, k;
    scanf("%d%d", &n, &k);
    for(int i=1; i<=k; i++)
        scanf("%d", a + i);
    if(!(k & 1))
    {
        int ans = 0;
        for(int i=1; i<=k; i+=2)
            ans += a[i + 1] - a[i];
        printf("%d\n", ans);
        return 0;
    }
    for(int i=1; i<k; i+=2)
        pre[i] = pre[i + 1] = pre[i - 1] + a[i + 1] - a[i];
    for(int i=k-1; i>0; i-=2)
        suf[i] = suf[i + 1] = suf[i + 2] + a[i + 1] - a[i];
    int ans = 1e9;
    for(int i=1; i<=k; i++)
    {
        int cur = i & 1? pre[i - 1] + suf[i + 1]
                    : a[i + 1] - a[i - 1] + pre[i - 2] + suf[i + 2];
        setmin(ans, cur);
    }
    printf("%d\n", ans);
    return 0;
}

#include <cstdio>
#include <unordered_map>
#include <set>
#include <atcoder/modint>
#define maxn 1005
using namespace std;

using modint = atcoder::modint998244353;

int n, m, fa[maxn * maxn];
char s[maxn][maxn];

int find(int x) { return fa[x] == x? fa[x]: fa[x] = find(fa[x]); }
inline int calc(int x, int y) { return x * m + y; }
inline int fc(int x, int y) { return find(calc(x, y)); }
inline void merge(int x, int y) { fa[find(x)] = find(y); }

int main()
{
    scanf("%d%d", &n, &m);
    for(int i=0; i<n; i++)
        scanf("%s", s[i]);
    int k = n * m;
    for(int i=0; i<k; i++)
        fa[i] = i;
    for(int i=0; i<n; i++)
        for(int j=0; j<m; j++)
        {
            if(s[i][j] != '#') continue;
            if(i && s[i - 1][j] == '#') merge(calc(i, j), calc(i - 1, j));
            if(j && s[i][j - 1] == '#') merge(calc(i, j), calc(i, j - 1));
        }
    int cnt = 0, tot = 0;
    for(int i=0; i<n; i++)
        for(int j=0; j<m; j++)
            if(s[i][j] == '#' && fc(i, j) == calc(i, j))
                cnt ++;
    modint ans = 0;
    for(int i=0; i<n; i++)
        for(int j=0; j<m; j++)
            if(s[i][j] == '.')
            {
                set<int> S;
                if(i && s[i - 1][j] == '#') S.insert(fc(i - 1, j));
                if(s[i + 1][j] == '#') S.insert(fc(i + 1, j));
                if(j && s[i][j - 1] == '#') S.insert(fc(i, j - 1));
                if(s[i][j + 1] == '#') S.insert(fc(i, j + 1));
                int cur = cnt - (int)S.size() + 1;
                ans += cur, tot ++;
            }
    printf("%d\n", (ans / tot).val());
    return 0;
}

inline void setmin(int& x, int y)
{
    if(y < x) x = y;
}

vector<int> G[maxn];

inline void add(int x, int y)
{
    G[x].push_back(y);
    G[y].push_back(x);
}

int root, low[maxn], cnt, dfn[maxn], ncut[maxn];

void tarjan(int v)
{
    low[v] = dfn[v] = ++cnt;
    ncut[v] = v != root;
    for(int u: G[v])
        if(!dfn[u])
        {
            tarjan(u);
            if(low[u] >= dfn[v])
                ncut[v] ++;
            setmin(low[v], low[u]);
        }
        else setmin(low[v], dfn[u]);
}

                id[i][j] = ++num;

    int cc = -1;

        if(!dfn[i])
            tarjan(root = i), cc ++;
