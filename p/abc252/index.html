#include <cstdio>
#define maxn 200005
using namespace std;

using LL = long long;
int cnt[maxn];

inline LL C2(int n) { return n * (n - 1LL) >> 1LL; }
inline LL C3(int n) { return C2(n) * (n - 2LL) / 3LL; }

int main()
{
    int n;
    scanf("%d", &n);
    for(int i=0; i<n; i++)
    {
        int a;
        scanf("%d", &a);
        cnt[a] ++;
    }
    LL ans = C3(n);
    for(int i=1; i<maxn; i++)
        if(cnt[i] > 1)
        {
            ans -= C2(cnt[i]) * (n - cnt[i]);
            if(cnt[i] > 2) ans -= C3(cnt[i]);
        }
    printf("%lld\n", ans);
    return 0;
}

#define maxn 200005
#define INF 9223372036854775807LL
using namespace std;

using LL = long long;
using pli = pair<LL, int>;

struct Edge
{
    int v, id; LL d;
    inline Edge(int u, int l, int i): v(u), d(l), id(i) {}
};

vector<Edge> G[maxn];
LL dis[maxn];
int ans[maxn];

int main()
{
    int n, m;
    scanf("%d%d", &n, &m);
    for(int i=1; i<=m; i++)
    {
        int a, b, c;
        scanf("%d%d%d", &a, &b, &c);
        G[--a].emplace_back(--b, c, i);
        G[b].emplace_back(a, c, i);
    }
    priority_queue<pli, vector<pli>, greater<pli>> q;
    for(int i=1; i<n; i++) dis[i] = INF;
    q.emplace(0LL, 0);
    while(!q.empty())
    {
        auto [d, v] = q.top(); q.pop();
        if(dis[v] == d)
            for(const Edge& e: G[v])

                LL nd = d + e.d;
                if(nd < dis[e.v])

                    ans[e.v] = e.id;
            }
    }

}

#include <queue>
using namespace std;

using LL = long long;

int main()
{
    int n;
    LL l;
    scanf("%d%lld", &n, &l);
    priority_queue<LL, vector<LL>, greater<LL>> q;
    for(int i=0; i<n; i++)
    {
        int x;
        scanf("%d", &x);
        q.push(x);
        l -= x;
    }
    if(l > 0) q.push(l);
    LL ans = 0LL;
    while(q.size() > 1)
    {
        LL x = q.top(); q.pop();
        x += q.top(); q.pop();
        ans += x;
        q.push(x);
    }
    printf("%lld\n", ans);
    return 0;
}

    #include <cstdio>
#define MOD 998244353
#define maxn 505
using namespace std;

using LL = long long;
int p[maxn], dp[maxn][maxn];

int main()
{
    int n;
    scanf("%d", &n);
    for(int i=0; i<n; i++)
        scanf("%d", p + i);
    for(int l=n; l>0; l--)
    {
        dp[l][l] = 1;
        for(int r=l+1; r<=n; r++)
        {
            dp[l][r] = dp[l + 1][r];
            for(int k=l+1; k<r; k++)
                if(p[l] < p[k] && (dp[l][r] += LL(dp[l + 1][k]) * dp[k][r] % MOD) >= MOD)
                    dp[l][r] -= MOD;
        }
    }
    printf("%d\n", dp[1][n]);
    return 0;
}
