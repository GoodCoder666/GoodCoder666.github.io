#include <iostream>
#include <atcoder/modint>
using namespace std;

using modint = atcoder::modint1000000007;

modint f(int n, int m)
{
    if(n < 0 || m < 0)
        return 0;
    modint ret = 1;
    for(int i=1; i<=m; i++)
        ret = ret * (n + i) / i;
    return ret;
}

int main()
{
    int n, m, k;
    scanf("%d%d%d", &n, &m, &k);
    if(n > m + k) puts("0");
    else printf("%d\n", (f(n, m) - f(n - k - 1, m + k + 1)).val());
    return 0;
}
