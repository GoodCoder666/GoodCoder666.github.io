#include <cstdio>
#include <vector>
#include <set>
#define maxn 100005
using namespace std;

vector<int> G[maxn];
bool used[maxn];
int color[maxn];

set<int> res;

void dfs(int v, int par)
{
    if(used[color[v]])
    {
        for(int u: G[v])
            if(u != par)
                dfs(u, v);
        return;
    }
    used[color[v]] = true;
    res.insert(v);
    for(int u: G[v])
        if(u != par)
            dfs(u, v);
    used[color[v]] = false;
}

int main()
{
    int n;
    scanf("%d", &n);
    for(int i=1; i<=n; i++)
        scanf("%d", color + i);
    while(--n)
    {
        int x, y;
        scanf("%d%d", &x, &y);
        G[x].push_back(y);
        G[y].push_back(x);
    }
    dfs(1, -1);
    for(int v: res)
        printf("%d\n", v);
    return 0;
}
